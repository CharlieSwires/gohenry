CREATE CACHED TABLE PUBLIC.PERSON(
    ID INT NOT NULL,
    TITLE VARCHAR(255),
    FIRST_NAME VARCHAR(255),
    LAST_NAME VARCHAR(255),
    EMAIL_ADDRESS VARCHAR(255),
    DOB DATE,
    GENDER VARCHAR(255),
    SECOND_NAME VARCHAR(255)
)
CREATE CACHED TABLE PUBLIC.CHILD(
    ID INT NOT NULL,
    PERSON_ID INT,
    PARENT_ID INT
)
